<Steps>
  <Step title="Create a new Next.js application">
    Run the following command to create a new Next.js application. Replace `my-protekt-app` with your project's name.

    ```bash Terminal
    npm create next-app@latest my-protekt-app -- --typescript --tailwind --eslint --app --src-dir --import-alias "@/*"
    ```

  </Step>

  <Step title="Install @protekt/nextjs">
    Navigate to your project directory and install the Protekt Next.js SDK:

    ```bash Terminal
    cd my-protekt-app
    npm install @protekt/nextjs
    ```

  </Step>

  <Step title="Add environment variables">
    Create a `.env.local` file in your project root and add your Protekt keys. You can find these in your **Protekt Dashboard**.

    ```bash .env.local

    NEXT_PUBLIC_PROTEKT_DOMAIN=your-app-domain.protekt.dev
    ```

  </Step>

  <Step title="Configure ProtektProvider">
    Protekt requires your application to be wrapped in the `<ProtektProvider>` component to provide authentication context.

    Update your `src/app/layout.tsx` file:

    ```tsx src/app/layout.tsx
    import type { Metadata } from 'next'
    import { ProtektProvider } from '@protekt/nextjs'
    import { Inter } from 'next/font/google'
    import './globals.css'

    const inter = Inter({ subsets: ['latin'] })

    export const metadata: Metadata = {
      title: 'My App with Protekt',
      description: 'Secure authentication powered by Protekt',
    }

    if (!process.env.NEXT_PUBLIC_PROTEKT_PUBLISHABLE_KEY) {
      throw new Error('Missing NEXT_PUBLIC_PROTEKT_PUBLISHABLE_KEY')
    }

    export default function RootLayout({
      children,
    }: {
      children: React.ReactNode
    }) {
      return (
        <ProtektProvider>
          <html lang="en">
            <body className={inter.className}>
              {children}
            </body>
          </html>
        </ProtektProvider>
      )
    }
    ```

  </Step>

  <Step title="Add middleware">
    Create a `middleware.ts` file in your project root to handle authentication across your app:
```ts middleware.ts
import { protektMiddleware } from '@protekt/nextjs/server'

export default protektMiddleware()

export const config = {
  matcher: [
    // Skip Next.js internals and static files
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    // Always run for API routes
    '/(api|trpc)(.*)',
  ],
}
```
  </Step>

  <Step title="Add authentication components">
    Replace the contents of `src/app/page.tsx` with authentication components that allow users to sign in and access their account:

    ```tsx src/app/page.tsx

      SignedIn,
      SignedOut,
      SignInButton,

      UserButton

      return (

            <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">

                      Sign up
                    </SignUpButton>
                  </SignedOut>

                  <SignedIn>
                    <UserButton />
                  </SignedIn>
                </div>
              </div>
            </div>
          </header>

          <main className="max-w-7xl mx-auto py-12 px-4 sm:px-6 lg:px-8">
            <SignedOut>
              <div className="text-center">
                <h2 className="text-3xl font-bold text-gray-900 mb-4">
                  Welcome to your secure app
                </h2>
                <p className="text-lg text-gray-600 mb-8">
                  Sign up or sign in to get started

                  You're successfully authenticated with Protekt.
                </p>
              </div>
            </SignedIn>
          </main>
        </div>
      )
    }
    ```

  </Step>

  <Step title="Test your integration">
    Start your development server:

    ```bash Terminal
    npm run dev
    ```

    Visit [http://localhost:3000](http://localhost:3000) in your browser. Click "Sign up" to create your first user and test the authentication flow.

  </Step>
