title: React Quickstart
---

<Steps>
  <Step title="Create a new React application">
    Run the following command to create a new React application. Replace `my-protekt-app` with your project's name.

    ```bash Terminal
    npm create react-app my-protekt-app --template typescript
    ```

  </Step>

  <Step title="Install @protekt/react">
    Navigate to your project directory and install the Protekt React SDK:

    ```bash Terminal
    cd my-protekt-app
    npm install @protekt/react
    ```

  </Step>

  <Step title="Add environment variables">
    Create a `.env.local` file in your project root and add your Protekt keys. You can find these in your **Protekt Dashboard**.

    ```bash .env.local
    REACT_APP_PROTEKT_PUBLISHABLE_KEY=pk_test_••••••••••••••••••••••••••••••••••
    REACT_APP_PROTEKT_DOMAIN=your-app-domain.protekt.dev
    ```

  </Step>

  <Step title="Configure ProtektProvider">
    Protekt requires your application to be wrapped in the `<ProtektProvider>` component to provide authentication context.

    Update your `src/index.tsx` file:

    ```tsx src/index.tsx
    import React from 'react'
    import ReactDOM from 'react-dom/client'
    import { ProtektProvider } from '@protekt/react'
    import App from './App'
    import './index.css'

    if (!process.env.REACT_APP_PROTEKT_PUBLISHABLE_KEY) {
      throw new Error('Missing REACT_APP_PROTEKT_PUBLISHABLE_KEY')
    }

    const root = ReactDOM.createRoot(
      document.getElementById('root') as HTMLElement
    )

    root.render(
      <React.StrictMode>
        <ProtektProvider
          publishableKey={process.env.REACT_APP_PROTEKT_PUBLISHABLE_KEY}
          domain={process.env.REACT_APP_PROTEKT_DOMAIN}
        >
          <App />
        </ProtektProvider>
      </React.StrictMode>
    )
    ```

  </Step>

  <Step title="Add authentication components">
    Replace the contents of `src/App.tsx` with authentication components that allow users to sign in and access their account:

    ```tsx src/App.tsx
    import {
      SignedIn,
      SignedOut,
      SignInButton,
      SignUpButton,
      UserButton
    } from '@protekt/react'

    function App() {
      return (
        <div className="min-h-screen bg-gray-50">
          <header className="bg-white shadow-sm border-b">
            <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
              <div className="flex justify-between items-center h-16">
                <h1 className="text-xl font-semibold text-gray-900">
                  My Protekt App
                </h1>

                <div className="flex items-center gap-4">
                  <SignedOut>
                    <SignInButton className="text-gray-600 hover:text-gray-900">
                      Sign in
                    </SignInButton>
                    <SignUpButton className="bg-blue-600 text-white px-4 py-2 rounded-md hover:bg-blue-700 transition-colors">
                      Sign up
                    </SignUpButton>
                  </SignedOut>

                  <SignedIn>
                    <UserButton />
                  </SignedIn>
                </div>
              </div>
            </div>
          </header>

          <main className="max-w-7xl mx-auto py-12 px-4 sm:px-6 lg:px-8">
            <SignedOut>
              <div className="text-center">
                <h2 className="text-3xl font-bold text-gray-900 mb-4">
                  Welcome to your secure app
                </h2>
                <p className="text-lg text-gray-600 mb-8">
                  Sign up or sign in to get started
                </p>
                <SignUpButton className="bg-blue-600 text-white px-8 py-3 rounded-md text-lg hover:bg-blue-700 transition-colors">
                  Get Started
                </SignUpButton>
              </div>
            </SignedOut>

            <SignedIn>
              <div className="text-center">
                <h2 className="text-3xl font-bold text-gray-900 mb-4">
                  Welcome back!
                </h2>
                <p className="text-lg text-gray-600">
                  You're successfully authenticated with Protekt.
                </p>
              </div>
            </SignedIn>
          </main>
        </div>
      )
    }

    export default App
    ```

  </Step>

  <Step title="Test your integration">
    Start your development server:

    ```bash Terminal
    npm start
    ```

    Visit [http://localhost:3000](http://localhost:3000) in your browser. Click "Sign up" to create your first user and test the authentication flow.

  </Step>

 
</Steps>
